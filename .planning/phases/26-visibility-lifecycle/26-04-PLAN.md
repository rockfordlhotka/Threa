@code {
    private List<Character>? archivedNpcs;
    private List<Table>? availableTables;
    private bool isLoading = true;
    private bool isProcessing;
    private string? errorMessage;
    private string? successMessage;

    // Restore modal state
    private bool showRestoreModal;
    private Character? restoreNpc;
    private Guid selectedTableId;

    // Delete modal state
    private bool showDeleteModal;
    private Character? deleteNpc;

    protected override async Task OnInitializedAsync()
    {
        await LoadArchivedNpcs();
    }

    private async Task LoadArchivedNpcs()
    {
        isLoading = true;
        try
        {
            archivedNpcs = await CharacterDal.GetArchivedNpcsAsync();
        }
        catch (Exception ex)
        {
            errorMessage = $"Failed to load archive: {ex.Message}";
        }
        finally
        {
            isLoading = false;
        }
    }

    private async Task ShowRestoreModal(Character npc)
    {
        restoreNpc = npc;
        selectedTableId = Guid.Empty;
        showRestoreModal = true;

        // Load available tables
        try
        {
            availableTables = await TableDal.GetActiveTablesAsync();
        }
        catch
        {
            availableTables = new List<Table>();
        }
    }

    private void CloseRestoreModal()
    {
        showRestoreModal = false;
        restoreNpc = null;
        availableTables = null;
    }

    private async Task ExecuteRestore()
    {
        if (restoreNpc == null || selectedTableId == Guid.Empty) return;
        isProcessing = true;

        try
        {
            // 1. Load character via portal for updating
            var character = await CharacterPortal.FetchAsync(restoreNpc.Id);

            // 2. Clear archive flag and set hidden (per CONTEXT.md)
            character.IsArchived = false;
            character.VisibleToPlayers = false; // Restored NPCs return hidden

            // 3. Save character
            await CharacterPortal.UpdateAsync(character);

            // 4. Attach to selected table
            var tableChar = new TableCharacter
            {
                TableId = selectedTableId,
                CharacterId = restoreNpc.Id,
                PlayerId = restoreNpc.PlayerId,
                JoinedAt = DateTime.UtcNow,
                ConnectionStatus = ConnectionStatus.Connected
            };
            await TableDal.AddCharacterToTableAsync(tableChar);

            successMessage = $"Restored {restoreNpc.Name} to table (hidden state)";
            CloseRestoreModal();
            await LoadArchivedNpcs();
        }
        catch (Exception ex)
        {
            errorMessage = $"Failed to restore: {ex.Message}";
        }
        finally
        {
            isProcessing = false;
        }
    }

    private void ConfirmDelete(Character npc)
    {
        deleteNpc = npc;
        showDeleteModal = true;
    }

    private void CloseDeleteModal()
    {
        showDeleteModal = false;
        deleteNpc = null;
    }

    private async Task ExecuteDelete()
    {
        if (deleteNpc == null) return;
        isProcessing = true;

        try
        {
            await CharacterDal.DeleteCharacterAsync(deleteNpc.Id);
            successMessage = $"Deleted {deleteNpc.Name} permanently";
            CloseDeleteModal();
            await LoadArchivedNpcs();
        }
        catch (Exception ex)
        {
            errorMessage = $"Failed to delete: {ex.Message}";
        }
        finally
        {
            isProcessing = false;
        }
    }

    private string GetDispositionBadgeClass(NpcDisposition disposition) => disposition switch
    {
        NpcDisposition.Hostile => "bg-danger",
        NpcDisposition.Neutral => "bg-secondary",
        NpcDisposition.Friendly => "bg-success",
        _ => "bg-secondary"
    };
}
