```razor
@page "/player/browse-campaigns"

@rendermode InteractiveServer

@attribute [Authorize]

@using Microsoft.AspNetCore.Authorization
@using Threa.Services
@using Threa.Client.Components.Shared
@using GameMechanics.Player
@using GameMechanics.GamePlay

@inject Csla.Blazor.State.StateManager StateManager
@inject RenderModeProvider RenderModeProvider
@inject NavigationManager NavigationManager
@inject ApplicationContext applicationContext
@inject IDataPortal<CharacterList> characterListPortal
@inject IDataPortal<TableList> tableListPortal
@inject IDataPortal<JoinRequestSubmitter> submitterPortal

<PageTitle>Browse Campaigns</PageTitle>

<div class="container" style="max-width: 900px;">
    @if (errorMessage != null)
    {
        <div class="alert alert-danger alert-dismissible fade show">
            @errorMessage
            <button type="button" class="btn-close" @onclick="() => errorMessage = null"></button>
        </div>
    }

    @if (successMessage != null)
    {
        <div class="alert alert-success alert-dismissible fade show">
            @successMessage
            <button type="button" class="btn-close" @onclick="() => successMessage = null"></button>
        </div>
    }

    @* Step 1: Character Selection *@
    @if (selectedCharacter == null)
    {
        <h3 class="mb-4">Select a Character to Join With</h3>

        @if (availableCharacters == null)
        {
            <div class="d-flex justify-content-center p-5">
                <div class="spinner-border text-primary" role="status">
                    <span class="visually-hidden">Loading characters...</span>
                </div>
            </div>
        }
        else if (!availableCharacters.Any())
        {
            @* No characters - redirect to creation (per CONTEXT.md) *@
            <div class="alert alert-info">
                <h5><i class="bi bi-info-circle me-2"></i>No Characters Available</h5>
                <p>You need to create a character before you can join a campaign.</p>
                <a href="/player/character" class="btn btn-primary">
                    <i class="bi bi-plus-lg me-1"></i>Create a Character
                </a>
            </div>
        }
        else
        {
            <div class="row row-cols-1 row-cols-md-2 g-3">
                @foreach (var character in availableCharacters.Where(c => c.IsPlayable && !c.IsAttachedToTable))
                {
                    <div class="col">
                        <div class="card h-100 character-select-card"
                             style="cursor: pointer;"
                             @onclick="() => SelectCharacter(character)">
                            <div class="card-body">
                                <h5 class="card-title">@character.Name</h5>
                                <p class="card-text text-muted">@character.Species</p>
                            </div>
                        </div>
                    </div>
                }
            </div>

            @if (!availableCharacters.Any(c => c.IsPlayable && !c.IsAttachedToTable))
            {
                <div class="alert alert-warning mt-3">
                    <h5><i class="bi bi-exclamation-triangle me-2"></i>No Available Characters</h5>
                    <p>All your characters are either not activated or already in a campaign.</p>
                    <a href="/player/characters" class="btn btn-secondary">View My Characters</a>
                </div>
            }
        }
    }
    else
    {
        @* Step 2: Campaign Browser *@
        <div class="d-flex justify-content-between align-items-center mb-4">
            <div>
                <h3 class="mb-0">Join a Campaign</h3>
                <small class="text-muted">
                    Playing as <strong>@selectedCharacter.Name</strong>
                    <button class="btn btn-link btn-sm p-0 ms-2" @onclick="ClearCharacterSelection">
                        (change)
                    </button>
                </small>
            </div>
        </div>

        @if (availableCampaigns == null)
        {
            <div class="d-flex justify-content-center p-5">
                <div class="spinner-border text-primary" role="status">
                    <span class="visually-hidden">Loading campaigns...</span>
                </div>
            </div>
        }
        else if (!availableCampaigns.Any())
        {
            <div class="card border-0 shadow-sm">
                <div class="card-body text-center py-5">
                    <i class="bi bi-journal-bookmark display-1 text-muted mb-3"></i>
                    <h5>No Campaigns Available</h5>
                    <p class="text-muted">Check back later!</p>
                </div>
            </div>
        }
        else
        {
            <div class="row row-cols-1 row-cols-md-2 row-cols-lg-3 g-4">
                @foreach (var campaign in availableCampaigns)
                {
                    <div class="col">
                        <div class="card h-100 campaign-card">
                            <div class="card-body">
                                <div class="d-flex justify-content-between align-items-start mb-2">
                                    <h5 class="card-title mb-0">@campaign.Name</h5>
                                    <ThemeIndicator Theme="@campaign.Theme" />
                                </div>
                                <p class="card-text text-muted small" style="min-height: 60px;">
                                    @(string.IsNullOrEmpty(campaign.Description)
                                        ? "No description provided."
                                        : (campaign.Description.Length > 120
                                            ? campaign.Description.Substring(0, 117) + "..."
                                            : campaign.Description))
                                </p>
                                <div class="d-flex justify-content-between align-items-center mt-3">
                                    <small class="text-muted">
                                        <i class="bi bi-person"></i> @GetGmDisplayName(campaign.GameMasterId)
                                    </small>
                                    <small class="text-muted">
                                        <i class="bi bi-people"></i> @GetPlayerCount(campaign.Id) players
                                    </small>
                                </div>
                            </div>
                            <div class="card-footer bg-transparent">
                                <button class="btn btn-primary w-100"
                                        disabled="@isSubmitting"
                                        @onclick="() => SubmitJoinRequest(campaign.Id)">
                                    @if (isSubmitting && submittingCampaignId == campaign.Id)
                                    {
                                        <span class="spinner-border spinner-border-sm me-1"></span>
                                    }
                                    Join with @selectedCharacter.Name
                                </button>
                            </div>
                        </div>
                    </div>
                }
            </div>
        }
    }
</div>

@code {
    private IEnumerable<CharacterInfo>? availableCharacters;
    private CharacterInfo? selectedCharacter;
    private IEnumerable<TableInfo>? availableCampaigns;
    private string? errorMessage;
    private string? successMessage;
    private bool isSubmitting;
    private Guid? submittingCampaignId;
    private int playerId;

    protected override async Task OnInitializedAsync()
    {
        await StateManager.InitializeAsync();
    }

    protected override async Task OnAfterRenderAsync(bool firstRender)
    {
        if (firstRender && RenderModeProvider.GetRenderMode(this).IsInteractive())
        {
            playerId = GetPlayerId();
            await LoadCharacters();
            StateHasChanged();
        }
    }

    private int GetPlayerId()
    {
        return int.Parse(applicationContext.Principal.Claims
            .First(c => c.Type == System.Security.Claims.ClaimTypes.NameIdentifier).Value);
    }

    private async Task LoadCharacters()
    {
        try
        {
            var list = await characterListPortal.FetchAsync(playerId);
            availableCharacters = list.AsEnumerable();
        }
        catch (Exception ex)
        {
            errorMessage = $"Failed to load characters: {ex.Message}";
        }
    }

    private async Task SelectCharacter(CharacterInfo character)
    {
        selectedCharacter = character;
        await LoadCampaigns();
    }

    private void ClearCharacterSelection()
    {
        selectedCharacter = null;
        availableCampaigns = null;
    }

    private async Task LoadCampaigns()
    {
        try
        {
            // Get all active tables (not ended)
            var list = await tableListPortal.FetchAsync();
            // Filter out tables where player is already GM
            availableCampaigns = list
                .Where(t => t.GameMasterId != playerId)
                .Where(t => t.Status != TableStatus.Ended)
                .OrderByDescending(t => t.CreatedAt)
                .AsEnumerable();
        }
        catch (Exception ex)
        {
            errorMessage = $"Failed to load campaigns: {ex.Message}";
        }
    }

    private async Task SubmitJoinRequest(Guid campaignId)
    {
        if (selectedCharacter == null) return;

        isSubmitting = true;
        submittingCampaignId = campaignId;
        errorMessage = null;

        try
        {
            var result = await submitterPortal.ExecuteAsync(
                selectedCharacter.Id,
                campaignId,
                playerId);

            if (result.Success)
            {
                // Navigate to pending requests page (per CONTEXT.md)
                NavigationManager.NavigateTo("/player/join-requests");
            }
            else
            {
                errorMessage = result.ErrorMessage;
            }
        }
        catch (Exception ex)
        {
            errorMessage = $"Failed to submit request: {ex.Message}";
        }
        finally
        {
            isSubmitting = false;
            submittingCampaignId = null;
        }
    }

    private string GetGmDisplayName(int gmId)
    {
        // For now, just show "GM" - could enhance to fetch actual name
        return "GM";
    }

    private int GetPlayerCount(Guid tableId)
    {
        // For now return 0 - could enhance with actual count
        return 0;
    }
}
```

```razor
    @if (errorMessage != null)
    {

    }

        </div>

                <a href="/player/browse-campaigns" class="btn btn-primary">
                    <i class="bi bi-search me-1"></i>Browse Campaigns
                </a>
            </div>
        </div>
    }
    else
    {
        <div class="list-group">
            @foreach (var request in requests.OrderByDescending(r => r.RequestedAt))
            {
                <div class="list-group-item">
                    <div class="d-flex justify-content-between align-items-start">
                        <div class="flex-grow-1">
                            <div class="d-flex align-items-center mb-1">
                                <strong class="me-2">@request.CharacterName</strong>
                                <i class="bi bi-arrow-right text-muted me-2"></i>
                                <span>@request.TableName</span>
                            </div>
                            <small class="text-muted">
                                Requested @request.RequestedAt.ToString("MMM d, yyyy 'at' h:mm tt")
                            </small>
                        </div>
                        <div class="text-end">
                            @if (request.Status == JoinRequestStatus.Pending)
                            {
                                <span class="badge bg-warning text-dark">
                                    <i class="bi bi-hourglass-split me-1"></i>Pending
                                </span>
                            }
                            else if (request.Status == JoinRequestStatus.Approved)
                            {
                                <div class="d-flex flex-column align-items-end gap-1">
                                    <span class="badge bg-success">
                                        <i class="bi bi-check-circle me-1"></i>Approved
                                    </span>
                                    <a href="/play/@request.TableId" class="btn btn-sm btn-success">
                                        <i class="bi bi-play-fill me-1"></i>Go to Campaign
                                    </a>
                                </div>
                            }
                            @* Denied requests are deleted, so no display needed *@
                        </div>
                    </div>
                </div>
            }
        </div>

        <div class="mt-3 text-muted small">
            <i class="bi bi-info-circle me-1"></i>
            Denied requests are automatically removed from this list.
        </div>
    }
</div>
```
